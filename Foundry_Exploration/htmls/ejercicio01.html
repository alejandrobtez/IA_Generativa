import os
from openai import AzureOpenAI
from dotenv import load_dotenv

load_dotenv()

raw_endpoint = os.getenv("AZURE_OPENAI_ENDPOINT").split("/openai")[0].rstrip("/")

client = AzureOpenAI(
    azure_endpoint=raw_endpoint,
    api_key=os.getenv("AZURE_OPENAI_API_KEY"),
    api_version="2024-08-01-preview"
)

deployment_name = os.getenv("AZURE_OPENAI_DEPLOYMENT_NAME")

messages = [{"role": "system", "content": "Eres un asistente educativo."}]

def chat():
    print("Escribe 'salir' para terminar.\n")
    
    while True:
        user_input = input("Usuario: ")
        
        if user_input.lower() in ['salir', 'exit', 'quit']:
            print("\nChat finalizado")
            break

        print(f"Tú: {user_input}")
        
        messages.append({"role": "user", "content": user_input})
        
        try:
            response = client.chat.completions.create(
                model=deployment_name,
                messages=messages
            )
            
            answer = response.choices[0].message.content
            
            print(f"Asistente: {answer}")
            print("-" * 30)
            
            messages.append({"role": "assistant", "content": answer})
            
        except Exception as e:
            print(f"Ocurrió un error: {e}")
            break

chat()

import json

def generar_json_estructurado():
    print("Respuesta estructurada en JSON:")
    
    tema_usuario = input("¿Sobre qué tema quieres generar un JSON? ")

    prompt_usuario = (
        f"Genera una lista sobre: {tema_usuario}. "
        "Para cada elemento, incluye al menos 3 campos descriptivos. "
        "Responde ÚNICAMENTE en formato JSON."
    )
    
    try:
        response = client.chat.completions.create(
            model=deployment_name,
            messages=[
                {"role": "system", "content": "Eres un asistente que solo responde en formato JSON."},
                {"role": "user", "content": prompt_usuario}
            ],

            response_format={ "type": "json_object" }
        )
        
        json_string = response.choices[0].message.content
        datos_validados = json.loads(json_string)

        print(f"Aquí tienes tu JSON sobre {tema_usuario}.")
        print(json.dumps(datos_validados, indent=4, ensure_ascii=False))

    except Exception as e:
        print(f"Error en la generación o validación JSON: {e}")

generar_json_estructurado()
